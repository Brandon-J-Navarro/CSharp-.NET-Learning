Algorithm
 - first number = get user input
     - if input is not numeric then print warning
 - second number = get user input
     - if input is not numeric then print warning
 - operation = get user input
     - lower case the operation for case insensitivity
     - if opertation is '+' or 'add' then
         print first number + second number
     - if operation is '-' or 'subtract' then
         print first number - second number
     - if operation is '*' or 'multiply' then
         print first number * second number
     - if operation is '/' or 'divide' then
         print first number / second number'
     - otherwise print warning
